local function PlayAnimation(id, startPos, stopDelay, speed)
    local anim = Instance.new("Animation")
    anim.AnimationId = "rbxassetid://" .. id
    local track = humanoid:LoadAnimation(anim)
    track:Play()
    track.TimePosition = startPos or 0
    track:AdjustSpeed(speed or 1)
    if stopDelay then
        task.spawn(function()
            task.wait(stopDelay)
            track:Stop()
        end)
    end
    return track
end

local nextImpactIndex = 1

local function PlayRandomDodgeSound()
    local snd = preloadedDodgeSounds[math.random(#preloadedDodgeSounds)]
    if snd.IsPlaying then snd:Stop() end
    snd:Play()

local function PlayNextImpactSound()
    local snd = preloadedImpactSounds[nextImpactIndex]
    if snd.IsPlaying then snd:Stop() end
    snd:Play()
    nextImpactIndex = nextImpactIndex + 1
    if nextImpactIndex > #preloadedImpactSounds then nextImpactIndex = 1 end
end

local lastDodgeAnimId = nil
local blockAnimId = "rbxassetid://10470389827"
local blockLoopActive = false

local function PickRandomDodgeAnim()
    if #dodgeAnimIds == 1 then return dodgeAnimIds[1] end
    local picked
    repeat
        picked = dodgeAnimIds[math.random(#dodgeAnimIds)]
    until picked ~= lastDodgeAnimId
    lastDodgeAnimId = picked
    return picked
end

local function PlayDodgeAnimation()
    local anim = Instance.new("Animation")
    anim.AnimationId = PickRandomDodgeAnim()
    for _, track in animator:GetPlayingAnimationTracks() do
        if table.find(dodgeAnimIds, track.Animation.AnimationId) or track.Animation.AnimationId == blockAnimId then
            track:Stop()
        end
    end
    local track = animator:LoadAnimation(anim)
    track.Priority = Enum.AnimationPriority.Action4
    track:Play()
    track:AdjustSpeed(Random.new():NextNumber(1, 1.5))
    PlayRandomDodgeSound()
    PlayNextImpactSound()
end

local prevBlockReact = character:GetAttribute("BlockReact")

local function CheckBlockReact()
    local isBlocking = character:GetAttribute("Blocking")
    local blockReact  = character:GetAttribute("BlockReact")
    if not blockReact then return end
    if isBlocking then
        if math.abs(math.abs(blockReact) - math.abs(prevBlockReact)) > 1 then
            PlayDodgeAnimation()
        elseif math.abs(blockReact) < 1 and not blockLoopActive then
            local anim = Instance.new("Animation")
            anim.AnimationId = blockAnimId
            local track = animator:LoadAnimation(anim)
            track.Looped = true
            track:Play()
            blockLoopActive = true
        end
    end
    prevBlockReact = blockReact
end

    end
end)

local function CacheSong(entry)
    local needsDownload = not isfile(entry.file)
    if not needsDownload then
        local ok, size = pcall(function() return #readfile(entry.file) end)
        if not ok or size < 10000 then
            needsDownload = true
        end
    end
    if needsDownload then
        local ok, data = pcall(game.HttpGet, game, entry.url)
        if ok and data and #data > 10000 then
            writefile(entry.file, data)
            return true
        else
            warn("[Music] Failed to download " .. entry.name)
            return false
        end
    end
    return true
end

local musicSound = Instance.new("Sound")
musicSound.Name   = "MusicLoop"
musicSound.Volume = musicVolume
musicSound.Looped = false
musicSound.Parent = Workspace

local musicActive      = true
local currentSongIndex = nil
local musicMaxAttempts = #musicSongs

local function FindSongIndexByName(name)
    for i, song in ipairs(musicSongs) do
        if song.name:lower() == name:lower() then
            return i
        end
    end
    return nil
end

local function PickNextSongIndex()
    if #musicSongs == 1 then return 1 end
    local index
    repeat
        index = math.random(#musicSongs)
    until index ~= currentSongIndex
    return index
end

local function WaitForSoundLength(sound, timeout)
    local startClock = os.clock()
    while sound.TimeLength <= 0 and os.clock() - startClock < (timeout or musicLoadTimeout) do
        task.wait()
    end
    return sound.TimeLength
end

local musicPlayToken   = 0
local usedStartingSong = false

local function PlayNextSong()
    if not musicActive then return end

    musicPlayToken = musicPlayToken + 1
    local myToken = musicPlayToken

    -- Loop instead of recursing on failure. Bounded by musicMaxAttempts so a bad run
    -- (dead download link, executor asset error, etc.) can't recurse forever or blow the stack.
    for attempt = 1, musicMaxAttempts do
        if musicPlayToken ~= myToken or not musicActive then return end

        if not usedStartingSong then
            usedStartingSong = true
            local startIndex = (typeof(_G.StartingSong) == "string" and _G.StartingSong ~= "") and FindSongIndexByName(_G.StartingSong)
            currentSongIndex = startIndex or PickNextSongIndex()
        else
            currentSongIndex = PickNextSongIndex()
        end
        local entry = musicSongs[currentSongIndex]

        musicSound:Stop()

        -- THE MAIN BUG: Sound.TimePosition does NOT reset itself when you change SoundId or
        -- call Stop(). A song that just played to completion leaves TimePosition sitting at
        -- (roughly) its own TimeLength. If the next song picked happens to be SHORTER than
        -- that leftover TimePosition, Play() starts already past the "end" of the new clip,
        -- so it immediately fires Ended again before you ever hear a sample of it. That reads
        -- exactly like "sometimes it just doesn't pick a new song" when really it's picking
        -- one and instantly skipping it.
        musicSound.TimePosition = 0

        if not CacheSong(entry) then
            continue
        end

        local getAsset = getcustomasset or getsynasset
        local ok, soundIdOrErr = pcall(getAsset, entry.file)
        if not ok or not soundIdOrErr then
            -- Executors cap how many custom assets can be registered at once; once that cap is
            -- hit (or the call errors for any other reason) this would previously throw all the
            -- way out of the Ended connection with no retry, silently killing that playback
            -- cycle. Now it's caught and just tries the next song instead.
            warn("[Music] getcustomasset failed for " .. entry.name .. ": " .. tostring(soundIdOrErr))
            continue
        end

        musicSound.SoundId = soundIdOrErr
        musicSound.TimePosition = 0 -- belt-and-suspenders: reset again after the new id is assigned

        local playOk, playErr = pcall(function() musicSound:Play() end)
        if not playOk then
            warn("[Music] Play() failed for " .. entry.name .. ": " .. tostring(playErr))
            continue
        end

        local length = WaitForSoundLength(musicSound)

        if musicPlayToken ~= myToken then return end

        if length <= 0 then
            warn("[Music] " .. entry.name .. " failed to load within " .. musicLoadTimeout .. "s, skipping")
            continue
        end

        print("[Music] Now playing: " .. entry.name)

        return -- success, we're done for this call
    end

    -- Every song in the list failed this round. Don't spin forever - back off and try again
    -- in a few seconds (covers a temporary network blip taking out every download at once).
    warn("[Music] All songs failed to play this round, retrying in 3s")
    task.delay(3, function()
        if musicPlayToken == myToken and musicActive then
            PlayNextSong()
        end
    end)
end

local function DoTeleport()
    local char = localPlayer.Character or localPlayer.CharacterAdded:Wait()
    local hrp  = char:WaitForChild("HumanoidRootPart")

    local allTargets = {}

    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= localPlayer and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
            table.insert(allTargets, plr.Character.HumanoidRootPart)
        end
    end

    local liveFolder = Workspace:FindFirstChild("Live")
    if liveFolder then
        for _, obj in ipairs(liveFolder:GetChildren()) do
            if obj.Name == "Weakest Dummy" and obj:FindFirstChild("HumanoidRootPart") then
                table.insert(allTargets, obj.HumanoidRootPart)
            end
        end
    end

    local nearestTarget = nil
    local closestDist   = math.huge
    for _, target in ipairs(allTargets) do
        local dist = (target.Position - hrp.Position).Magnitude
        if dist < closestDist then
            closestDist   = dist
            nearestTarget = target
        end
    end

    local originalTransparency = {}
    local function StoreTransparency(obj)
        if obj:IsA("BasePart") or obj:IsA("Decal") or obj:IsA("Texture") then
            originalTransparency[obj] = obj.Transparency
        end
        for _, child in ipairs(obj:GetChildren()) do
            StoreTransparency(child)
        end
    end
    StoreTransparency(char)

    local function SetVisibility(hide)
        for obj, trans in pairs(originalTransparency) do
            if trans < 1 then
                obj.Transparency = hide and 1 or trans
            end
        end
    end

    PlayAnimation("120577018823573", 0, 0.5, 2)
    task.wait(0.5)

    EmitVFX(ReplicatedStorage.Resources.NinjaUlt.Start_Effects.Particles.Low)
    EmitVFX(ReplicatedStorage.Resources.NinjaUlt.Start_Effects.Particles.Center_Tilted)
    EmitVFX(ReplicatedStorage.Resources.NinjaUlt.Start_Effects.Particles.Ground)

    local teleportSfx = Instance.new("Sound")
    teleportSfx.SoundId = "rbxassetid://3339374157"
    teleportSfx.Volume  = 1
    teleportSfx.Parent  = hrp
    teleportSfx:Play()

    if nearestTarget then
        local dir = (nearestTarget.Position - hrp.Position).Unit
        hrp.CFrame = CFrame.new(nearestTarget.Position - dir * -3, nearestTarget.Position)
    end

    SetVisibility(true)  task.wait(0.01)
    SetVisibility(false) task.wait(0.01)
    SetVisibility(true)  task.wait(0.01)
    SetVisibility(false)
end

local function SpawnHitbox(onHit)
    local hitbox = Instance.new("Part")
    hitbox.Name        = "Hitbox"
    hitbox.Size        = Vector3.new(10, 8, 10)
    hitbox.Anchored    = true
    hitbox.CanCollide  = false
    hitbox.Transparency = 1
    hitbox.CFrame      = rootPart.CFrame * CFrame.new(0, 0, -5)
    hitbox.Parent      = Workspace

    local hitHandled = false

    hitbox.Touched:Connect(function(touchedPart)
        if hitHandled then return end
        local model = touchedPart:FindFirstAncestorOfClass("Model")
        if not model or model == character then return end
        local targetHumanoid = model:FindFirstChild("Humanoid")
        if not targetHumanoid then return end

        hitHandled = true
        hitbox:Destroy()
        onHit(model)
    end)

    Debris:AddItem(hitbox, 0.15)
end

local function updateToolNames()
    for _, slot in ipairs(hotbar:GetChildren()) do
        if slot:FindFirstChild("Base") and slot.Base:FindFirstChild("ToolName") then
            local toolName = slot.Base.ToolName
            local newName  = toolNameMap[toolName.Text]
            if newName then
                if not slot:FindFirstChild("CustomName") then
                    toolName.Visible = false
                    local clone = toolName:Clone()
                    clone.Name    = "CustomName"
                    clone.Parent  = slot
                    clone.Text    = newName
                    clone.Visible = true
                elseif slot.CustomName.Text ~= newName then
                    toolName.Visible     = false
                    slot.CustomName.Text = newName
                end
            end
        end
    end
end

local function ApplyMarketplaceClothing(assetId, clothingType)
    local ok, objects = pcall(game.GetObjects, game, "rbxassetid://" .. assetId)
    if not ok or not objects then return end
    for _, obj in ipairs(objects) do
        local clothing = obj:IsA(clothingType) and obj or obj:FindFirstChildWhichIsA(clothingType)
        if clothing then
            local existing = character:FindFirstChildOfClass(clothingType)
            if existing then existing:Destroy() end
            local newCloth = Instance.new(clothingType)
            if clothingType == "Shirt" then
                newCloth.ShirtTemplate = clothing.ShirtTemplate
            else
                newCloth.PantsTemplate = clothing.PantsTemplate
            end
            newCloth.Parent = character
            break
        end
    end
end
